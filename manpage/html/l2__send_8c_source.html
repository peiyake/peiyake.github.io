211 uint16_t checksum (uint16_t *addr, int len)
212 {
213  int count = len;
214  register uint32_t sum = 0;
215  uint16_t answer = 0;
216 
217  // Sum up 2-byte values until none or only one byte left.
218  while (count > 1) {
219  sum += *(addr++);
220  count -= 2;
221  }
222 
223  // Add left-over byte, if any.
224  if (count > 0) {
225  sum += *(uint8_t *) addr;
226  }
227 
228  // Fold 32-bit sum into 16 bits; we lose information by doing this,
229  // increasing the chances of a collision.
230  // sum = (lower 16 bits) + (upper 16 bits shifted right 16 bits)
231  while (sum >> 16) {
232  sum = (sum & 0xffff) + (sum >> 16);
233  }
234 
235  // Checksum is one's compliment of sum.
236  answer = ~sum;
237 
238  return (answer);
239 }

241 uint16_t udp4_checksum (struct iphdr *iph, struct udphdr *udph, uint8_t *payload, int payloadlen)
242 {
243  char buf[1500] = {0};
244  char *ptr;
245  int chksumlen = 0;
246  int i;
247 
248  ptr = &buf[0]; // ptr points to beginning of buffer buf
249 
250  /* 构造 伪首部 */
251  // Copy source IP address into buf (32 bits)
252  memcpy (ptr, &iph->saddr, sizeof (iph->saddr));
253  ptr += sizeof (iph->saddr);
254  chksumlen += sizeof (iph->saddr);
255  // Copy destination IP address into buf (32 bits)
256  memcpy (ptr, &iph->daddr, sizeof (iph->daddr));
257  ptr += sizeof (iph->daddr);
258  chksumlen += sizeof (iph->daddr);
259  // Copy zero field to buf (8 bits)
260  *ptr = 0; ptr++;
261  chksumlen += 1;
262  // Copy transport layer protocol to buf (8 bits)
263  memcpy (ptr, &iph->protocol, sizeof (iph->protocol));
264  ptr += sizeof (iph->protocol);
265  chksumlen += sizeof (iph->protocol);
266  // Copy UDP length to buf (16 bits)
267  memcpy (ptr, &udph->len, sizeof (udph->len));
268  ptr += sizeof (udph->len);
269  chksumlen += sizeof (udph->len);
270 
271  /* 真实UDP首部 */
272  // Copy UDP source port to buf (16 bits)
273  memcpy (ptr, &udph->source, sizeof (udph->source));
274  ptr += sizeof (udph->source);
275  chksumlen += sizeof (udph->source);
276  // Copy UDP destination port to buf (16 bits)
277  memcpy (ptr, &udph->dest, sizeof (udph->dest));
278  ptr += sizeof (udph->dest);
279  chksumlen += sizeof (udph->dest);
280  // Copy UDP length again to buf (16 bits)
281  memcpy (ptr, &udph->len, sizeof (udph->len));
282  ptr += sizeof (udph->len);
283  chksumlen += sizeof (udph->len);
284  // Copy UDP checksum to buf (16 bits)
285  // Zero, since we don't know it yet
286  *ptr = 0; ptr++;
287  *ptr = 0; ptr++;
288  chksumlen += 2;
289 
290  /* 数据部分 */
291  // Copy payload to buf
292  memcpy (ptr, payload, payloadlen);
293  ptr += payloadlen;
294  chksumlen += payloadlen;
295 
296  // Pad to the next 16-bit boundary
297  for (i=0; i<payloadlen%2; i++, ptr++) {
298  *ptr = 0;
299  ptr++;
300  chksumlen++;
301  }
302 
303  return checksum ((uint16_t *) buf, chksumlen);
304 }
